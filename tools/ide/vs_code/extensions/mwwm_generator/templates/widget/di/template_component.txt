import 'package:flutter/material.dart';
import 'package:flutter_template/ui/base/di/widget_component.dart';
import 'package:injector/injector.dart';

/// [Component] для [Template]
class TemplateComponent extends WidgetComponent {
  TemplateComponent(BuildContext context) : super(context);
}

/// Билдер для [GiftsWidgetModel]
TemplateWidgetModel createTemplateWidgetModel(BuildContext context) {
  var component = Injector.of<TemplateComponent>(context).component;
  return TemplateWidgetModel(
    component.wmDependencies,
    component.navigator
  );
}
